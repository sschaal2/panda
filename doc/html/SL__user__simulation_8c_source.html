112  // simulate the computed F/T sensor of external forces. We can use the
113  // ucontact[J7] wrench info for this, which is in world coordinates.
114  // Transform into J7_LINK, which identical to J7 joint coordinates
115 
116  for (i=1; i<=N_CART; ++i) {
117  force_local[i] = torque_local[i] = 0;
118  for (j=1; j<=N_CART; ++j) {
119  force_local[i] += Alink_sim[J7_LINK][j][i] * ucontact[J7].f[j];
120  torque_local[i] += Alink_sim[J7_LINK][j][i] * ucontact[J7].t[j];
121  }
122  }
123 
124 
125  // this force/torque is slighly wrong: it needs to be at the loadcell coordinate
126  // system, and needs to take into account moments from translating from
127  // J7 to the load cell coordinate. Thus, the moments need to be corrected.
128 
129  vec_zero_size(offset,N_CART);
130  offset[_Z_] = FL+FT_OFF_Z;
131  vec_mult_outer_size(offset, force_local, N_CART, torque_correction);
132 
133  for (i=1; i<=N_CART; ++i) {
134  torque_local[i] -= torque_correction[i];
135  }
136 
137  // the computed torque is reported in load cell coordinates.
138  // Thus, we need to transform back into the world coordinate system ...
139 
140  for (i=1; i<=N_CART; ++i) {
141  force_world[i] = torque_world[i] = 0;
142  for (j=1; j<=N_CART; ++j) {
143  force_world[i] += Alink_sim[J7_LINK][i][j] * force_local[j];
144  torque_world[i] += Alink_sim[J7_LINK][i][j] * torque_local[j];
145  }
146  }
147 
148  // ... and now to endeffector coordinates, which has the same orientation
149  // as the load cell
150  for (i=1; i<=N_CART; ++i) {
151  force_local[i] = torque_local[i] = 0;
152  for (j=1; j<=N_CART; ++j) {
153  force_local[i] += Alink_sim[FLANGE][j][i] * force_world[j];
154  torque_local[i] += Alink_sim[FLANGE][j][i] * torque_world[j];
155  }
156  }
157

161  }
